-- Load UI Library (Rayfield)
local Nocturne = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

-- Window Utama
local MainUI = Nocturne:CreateWindow({
    Name = "Nocturne Premium",
    LoadingTitle = "Loading...",
    LoadingSubtitle = "by Kaizel",
    ConfigurationSaving = { Enabled = false }
})

-- AUTO FISH
local AutoFishTab = MainUI:CreateTab("Auto Fish", 4483362458)
AutoFishTab:CreateToggle({
    Name = "Enable Auto Fish",
    CurrentValue = false,
    Flag = "AutoFish",
    Callback = function(Value)
        print("Auto Fish: "..tostring(Value))
        -- isi script auto fish di sini
    end
})

    end
})

-- PLAYER MENU
local PlayerTab = MainUI:CreateTab("Player", 4483362458)
PlayerTab:CreateSlider({
    Name = "Walk Speed",
    Range = {16, 200},
    Increment = 1,
    CurrentValue = 16,
    Callback = function(Value)
        game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = Value
    end
})
PlayerTab:CreateToggle({
    Name = "Infinite Jump",
    CurrentValue = false,
    Callback = function(Value)
        print("Infinite Jump: "..tostring(Value))
        -- isi script infinite jump di sini
    end
})
